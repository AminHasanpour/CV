% ======================================================================
% SECTION ORDER
% Define the order in which sections appear in the CV
% To change the order, modify the \rendersections command below
% ======================================================================

\newcommand{\rendersections}{
    % Header (typically always first)
    \ifshowheader
        \input{sections/header.tex}
    \fi
    
    % Summary
    \ifshowsummary
        \input{sections/summary.tex}
    \fi
    
    % Education
    \ifshoweducation
        \input{sections/education.tex}
    \fi
    
    % Domain Expertise
    \ifshowdomainexpertise
        \input{sections/domain-expertise.tex}
    \fi
    
    % Skills
    \ifshowskills
        \input{sections/skills.tex}
    \fi
    
    % Publications
    \ifshowpublications
        \input{sections/publications.tex}
    \fi
    
    % Work Experience
    \ifshowworks
        \input{sections/work-experience.tex}
    \fi
    
    % Projects
    \ifshowprojects
        \input{sections/projects.tex}
    \fi
    
    % Teaching Experience
    \ifshowteaching
        \input{sections/teaching.tex}
    \fi
    
    % Achievements and Awards
    \ifshowachievements
        \input{sections/achievements.tex}
    \fi
    
    % Personal Highlights
    \ifshowpersonal
        \input{sections/personal.tex}
    \fi
}
