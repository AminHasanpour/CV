% ======================================================================
% MULTI-COLUMN ENTRY COMMAND
% ======================================================================
% A flexible command for creating entries with arbitrary number of columns
% Usage: \multicolentry{<column_specs>}{<content_1>}{<content_2>}...{<content_n>}
% where <column_specs> is comma-separated list like "fill,4.5cm" or "2cm,fill,3cm"
% Use "fill" or "*" for variable-width columns that fill remaining space

\ExplSyntaxOn

% Helper to count columns in spec
\seq_new:N \l_multicolentry_spec_seq
\int_new:N \l_multicolentry_numcols_int
\tl_new:N \l_multicolentry_spec_tl

% Parse column spec and convert to paracol format
\cs_new:Npn \__multicolentry_parse_spec:n #1
{
    \seq_set_split:Nnn \l_multicolentry_spec_seq { , } { #1 }
    \int_set:Nn \l_multicolentry_numcols_int { \seq_count:N \l_multicolentry_spec_seq }
    
    % Convert spec: replace "fill" or "*" with "\fill"
    \tl_clear:N \l_multicolentry_spec_tl
    \seq_map_inline:Nn \l_multicolentry_spec_seq
    {
        \tl_if_empty:NF \l_multicolentry_spec_tl
        {
            \tl_put_right:Nn \l_multicolentry_spec_tl { , }
        }
        \str_if_eq:nnTF { ##1 } { fill }
        {
            \tl_put_right:Nn \l_multicolentry_spec_tl { \fill }
        }
        {
            \str_if_eq:nnTF { ##1 } { * }
            {
                \tl_put_right:Nn \l_multicolentry_spec_tl { \fill }
            }
            {
                \tl_put_right:Nn \l_multicolentry_spec_tl { ##1 }
            }
        }
    }
}

% Main multicolentry command
\NewDocumentCommand{\multicolentry}{ m }
{
    \__multicolentry_parse_spec:n { #1 }
    \__multicolentry_build:n { \int_use:N \l_multicolentry_numcols_int }
}

% Build the entry by collecting remaining arguments
\cs_new:Npn \__multicolentry_build:n #1
{
    \int_compare:nNnTF { #1 } = { 2 }
    {
        \__multicolentry_two:nn
    }
    {
        \int_compare:nNnTF { #1 } = { 3 }
        {
            \__multicolentry_three:nnn
        }
        {
            \int_compare:nNnTF { #1 } = { 4 }
            {
                \__multicolentry_four:nnnn
            }
            {
                \int_compare:nNnTF { #1 } = { 5 }
                {
                    \__multicolentry_five:nnnnn
                }
                {
                    \msg_error:nn { multicolentry } { unsupported-columns }
                }
            }
        }
    }
}

% Two column version
\NewDocumentCommand{\__multicolentry_two:nn}{ +m +m }
{
    \begin{onecolentry}
        \exp_args:NV \setcolumnwidth \l_multicolentry_spec_tl
        \begin{paracol}{2}
            #1
            \switchcolumn
            #2
        \end{paracol}
    \end{onecolentry}
}
