% One column entry
\newenvironment{onecolentry}{
    \begin{adjustwidth}{
        0.2 cm + 0.00001 cm
    }{
        0.2 cm + 0.00001 cm
    }
}{
    \end{adjustwidth}
}

% Two column entry
\newenvironment{twocolentry}[2][]{
    \onecolentry
    % Use optional width if given, else default
    \ifx&#1&%
        \setlength{\twocolrightwidth}{4.5cm}% default
    \else
        \setlength{\twocolrightwidth}{#1}% user-specified
    \fi
    \def\secondColumn{#2}
    \setcolumnwidth{\fill, \twocolrightwidth}
    \begin{paracol}{2}
}{
    \switchcolumn \raggedleft \secondColumn
    \end{paracol}
    \endonecolentry
}

% Two column entry with left fixed
\newenvironment{twocolentryleftfixed}[1]{
    \onecolentry
    \def\firstColumn{#1}
    \setcolumnwidth{2.5cm, \fill}
    \begin{paracol}{2}
    {\raggedright \firstColumn} \switchcolumn
}{
    \end{paracol}
    \endonecolentry
}

% Three column entry
\newenvironment{threecolentry}[3][]{
    \onecolentry
    \def\thirdColumn{#3}
    \setcolumnwidth{1.2 cm, \fill, 4.5 cm}
    \begin{paracol}{3}
    {\raggedright #2} \switchcolumn
}{
    \switchcolumn \raggedleft \thirdColumn
    \end{paracol}
    \endonecolentry
}
